# Команды для выполнения в PowerShell от имени администратора:

# 1. Включить Hyper-V (если доступно)
Enable-WindowsOptionalFeature -Online -FeatureName Microsoft-Hyper-V -All -NoRestart

# 2. Включить платформу виртуальной машины
Enable-WindowsOptionalFeature -Online -FeatureName VirtualMachinePlatform -NoRestart

# 3. Включить подсистему Windows для Linux
Enable-WindowsOptionalFeature -Online -FeatureName Microsoft-Windows-Subsystem-Linux -NoRestart

# После выполнения команд перезагрузите компьютер
# Затем установите WSL2: wsl --install
